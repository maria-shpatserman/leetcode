class Solution {
    public int maxProfit(int[] prices) {
        int min = prices[0];
        int maxDiff = 0;
        if(prices.length == 1) return maxDiff;
        for(int i=1;i<prices.length;i++){
            if(prices[i]<min){
                min=prices[i];
            }
            else {
                maxDiff= Math.max((prices[i] - min), maxDiff);
            }
        }

        return maxDiff;
    }
}
